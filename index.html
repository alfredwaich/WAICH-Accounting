Conectar Google Sheets
Sigue estos 3 pasos UNA SOLA VEZ. Después cada proyecto se guarda automáticamente en su propia hoja.
✅ Conectado a Google Sheets. Cada proyecto se guarda en su propia hoja automáticamente.
1
Crea un Google Sheet nuevo
Abre una pestaña nueva, ve a sheets.new y nómbralo Waich Construction Bookkeeper.
📊 Abrir sheets.new
2
Pega este código en Extensions → Apps Script
En tu Google Sheet ve a Extensions → Apps Script. Borra todo (Ctrl+A, Delete), pega el código de abajo. Luego: Deploy → New deployment → Web app → "Who has access": Anyone → Deploy → autoriza con Gmail → copia la URL.
// WAICH CONSTRUCTION — Google Apps Script Backend const HEADERS = ['Nro','Date','Payee','Debit','Credit','Category','Check/Ref','Bank Account','Payment Method','Notes','Timestamp']; function getOrCreateSheet(ss, name) { let s = ss.getSheetByName(name); if (!s) { s = ss.insertSheet(name); s.appendRow(HEADERS); s.getRange(1,1,1,HEADERS.length).setFontWeight('bold').setBackground('#0d2045').setFontColor('#ffffff'); s.setFrozenRows(1); } return s; } function doPost(e) { try { const ss = SpreadsheetApp.getActiveSpreadsheet(); const data = JSON.parse(e.postData.contents); const sheetName = data.project || 'Transactions'; const sheet = getOrCreateSheet(ss, sheetName); if (data.action === 'add') { sheet.appendRow([data.nro, data.date, data.payee, data.debit||'', data.credit||'', data.category, data.check||'', data.bank, data.payment, data.notes||'', new Date().toISOString()]); return ok('Row added'); } if (data.action === 'bulk_delete') { const ids = data.ids.map(String); const rows = sheet.getDataRange().getValues(); for (let i = rows.length-1; i >= 1; i--) { if (ids.includes(String(rows[i][0]))) sheet.deleteRow(i+1); } return ok('Deleted'); } if (data.action === 'bulk') { sheet.clearContents(); sheet.appendRow(HEADERS); sheet.getRange(1,1,1,HEADERS.length).setFontWeight('bold').setBackground('#0d2045').setFontColor('#ffffff'); if (data.rows.length) { const vals = data.rows.map(r => [r.nro, r.date, r.payee, r.debit||'', r.credit||'', r.category, r.check||'', r.bank, r.payment, r.notes||'', '']); sheet.getRange(2,1,vals.length,HEADERS.length).setValues(vals); } return ok('Bulk done'); } } catch(err) { return err_(err); } } function doGet(e) { try { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheetName = e.parameter.project || 'Transactions'; const sheet = ss.getSheetByName(sheetName); if (!sheet) return ContentService.createTextOutput(JSON.stringify({status:'ok',rows:[]})).setMimeType(ContentService.MimeType.JSON); const data = sheet.getDataRange().getValues(); const headers = data[0]; const rows = data.slice(1).filter(r=>r.some(c=>c!=='')).map(r=>{ const o={}; headers.forEach((h,i)=>o[h]=r[i]); return o; }); return ContentService.createTextOutput(JSON.stringify({status:'ok',rows})).setMimeType(ContentService.MimeType.JSON); } catch(err) { return err_(err); } } const ok = msg => ContentService.createTextOutput(JSON.stringify({status:'ok',message:msg})).setMimeType(ContentService.MimeType.JSON); const err_ = e => ContentService.createTextOutput(JSON.stringify({status:'error',message:e.toString()})).setMimeType(ContentService.MimeType.JSON);
